# win 10安装python2

```shell
# 下载安装
https://www.python.org/downloads/release/python-2716/

# 安装pip
# 下载get-pip
https://bootstrap.pypa.io/get-pip.py
# 安装
D:/program/program_code/python27/python.exe "c:/Users/CRAB/Desktop/python27/get-pip.py"

# 安装virtualenv
D:/program/program_code/python27/Scripts/pip2.7.exe install virtualenv\

# 创建虚拟环境
cd D:/program/program_code/python27_virtualenv_home
D:/program/program_code/python27/Scripts/virtualenv.exe --clear img_overlap

# 激活
cd D:/program/program_code/python27_virtualenv_home/img_overlap/Scripts
activate
```
